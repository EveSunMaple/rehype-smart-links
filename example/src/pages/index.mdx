<h2 class="text-center text-3xl font-bold mb-10">在 Astro 中使用</h2>

<div class="card bg-base-100 shadow-xl border border-base-300 mb-16">
  <div class="card-body">
    <h3 class="card-title mb-4">在你的 `astro.config.mjs` 文件中添加插件：</h3>
    
    ```js
    import { defineConfig } from 'astro/config';
    import rehypeSmartLinks from 'rehype-smart-links';

    export default defineConfig({
      markdown: {
        rehypePlugins: [[rehypeSmartLinks, {
          // 你的配置选项
          internalLinkClass: 'internal-link',
          brokenLinkClass: 'broken-link',
          externalLinkClass: 'external-link'
        }]]
      }
    });
    ```
  </div>
</div>
